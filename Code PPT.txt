        % Button pushed function: ChiaDoiButton
        function ChiaDoiButtonPushed(app, event)
            f = app.PhuongTrinhEditField.Value;
            a = app.KhoangPhanLyAEditField.Value;
            b = app.KhoangPhanLyBEditField.Value;
            saiso = app.SaiSoEditField.Value;
            
            f = str2func(['@(x)', f]);

            function [x, n] = chiadoi(fx, a, b, saiso)
                n = 0;
                while (1)
                    c = (a+b)/2;
                    fac = fx(a) * fx(c);
                    if fac<0
                        b = c;
                    else 
                        a = c;
                    end
                    e = abs(b-a);
                    if e < saiso
                        break;
                    end
                    n = n+1;
                end
            x = c;
            end

            [x, solanlap] = chiadoi(f, a, b, saiso);
            app.KetQuaNghiemLabel.Text = sprintf('Kết quả nghiệm: %f', x);
            app.SoLanLapLabel.Text = sprintf('Số lần lặp: %d', solanlap);

            cla(app.UIAxes);
            x_plot = linspace(a, b, 100);
            y_plot = f(x_plot);
            plot(app.UIAxes, x_plot, y_plot, 'b', 'LineWidth', 1.5);
            hold(app.UIAxes, 'on');
            yline(app.UIAxes, 0, 'r', 'LineWidth', 1.5);
            plot(app.UIAxes, x, 0, 'go', 'MarkerSize', 7, 'MarkerFaceColor', 'g');
            hold(app.UIAxes, 'off');
            grid(app.UIAxes, 'on');
            title(app.UIAxes, 'Tìm nghiệm');
            legend(app.UIAxes, 'Đồ thị hàm số', 'y = 0');
        end

        % Button pushed function: LapButton
        function LapButtonPushed(app, event)
            
            f = app.PhuongTrinhEditField.Value;
            a = app.KhoangPhanLyAEditField.Value;
            b = app.KhoangPhanLyBEditField.Value;
            saiso = app.SaiSoEditField.Value;
            
            f = str2func(['@(x)', f]);

            function g_func = convertFunction(f)
                % Hàm chuyển đổi phương trình 0 = f(x) thành x = g(x)       
                syms x; 
                f_sym = f(x); 
                rearranged = solve(f_sym == 0, x); % Giải phương trình 0 = f(x)
                
                % Chuyển biểu thức ký hiệu thành hàm ẩn danh
                g_func = matlabFunction(rearranged, 'Vars', x);
            end

            function [x, n] = lap(fx, fp, a, b, saiso)
            n = 0;
            x0 = b;
            while (1)
                x1 = fp(x0);
                if abs(x1 - x0) < saiso
                    break;
                end
                n = n+1;
                x0 = x1;
            end
            x = x1;
            end
            
            fp = convertFunction(f);
            [x, solanlap] = lap(f, fp, a, b, saiso);
            app.KetQuaNghiemLabel.Text = sprintf('Kết quả nghiệm: %f', x);
            app.SoLanLapLabel.Text = sprintf('Số lần lặp: %d', solanlap);

            cla(app.UIAxes);
            x_plot = linspace(a, b, 100);
            y_plot = f(x_plot);
            plot(app.UIAxes, x_plot, y_plot, 'b', 'LineWidth', 1.5);
            hold(app.UIAxes, 'on');
            yline(app.UIAxes, 0, 'r', 'LineWidth', 1.5);
            plot(app.UIAxes, x, 0, 'go', 'MarkerSize', 7, 'MarkerFaceColor', 'g');
            hold(app.UIAxes, 'off');
            grid(app.UIAxes, 'on');
            title(app.UIAxes, 'Tìm nghiệm');
            legend(app.UIAxes, 'Đồ thị hàm số', 'y = 0');
        end

        % Button pushed function: LagrangeButton
        function LagrangeButtonPushed(app, event)
            xa = app.X_EditField.Value;
            ya = app.Y_EditField.Value;
            xa = str2num(xa);
            ya = str2num(ya);
            x1 = app.GiaTriNoiSuyEditField.Value;

            function result = Lagrange(xa, ya, x)
                n = length(xa);
                sum = 0;
                for i = 1:n
                    product = ya(i);
                    for j = 1:n
                        if i ~= j
                            product = product * (x - xa(j)) / (xa(i) - xa(j));
                        end
                    end
                    sum = sum + product;
                end
                result = sum;
            end

            KquaNS = Lagrange(xa, ya, x1);

            x = sym('x');
            DaThucNS = Lagrange(xa, ya, x);
            DaThucNS = vpa(simplify(DaThucNS), 4); 

            app.KetQuaNoiSuyLabel.Text = sprintf('Kết quả nội suy: %.4f', KquaNS);
            app.KetQuaDaThucNoiSuyLabel.Text = sprintf('Đa thức nội suy: %s', char(DaThucNS));

            cla(app.UIAxesNS);
            x_vals = linspace(min(xa), max(xa), 100); 
            y_vals = arrayfun(@(x_val) Lagrange(xa, ya, x_val), x_vals); 
            plot(app.UIAxesNS, xa, ya, 'ro', 'MarkerFaceColor', 'r'); % Dữ liệu thực với điểm đỏ
            hold(app.UIAxesNS, 'on');
            plot(app.UIAxesNS, x_vals, y_vals, 'b-', 'LineWidth', 1.5); % Đồ thị hàm nội suy với đường xanh
            grid(app.UIAxesNS, 'on');
            hold(app.UIAxesNS, 'off'); 
            legend(app.UIAxesNS, 'Dữ liệu thực', 'Hàm nội suy');
        end

        % Button pushed function: NewtonButton_
        function NewtonButton_Pushed(app, event)
            xa = app.X_EditField.Value;
            ya = app.Y_EditField.Value;
            xa = str2num(xa);
            ya = str2num(ya);
            x1 = app.GiaTriNoiSuyEditField.Value;

            function d = DividedDifference(xa, ya)
              n = length(xa);
              d = ya;
              for i = 2:n
                for j = n:-1:i
                   d(j) = (d(j) - d(j-1)) / (xa(j) - xa(j-i+1));
                end
              end
            end

            function result = NewtonForm(xa, da, x)
                n = length(da);  
                result = da(n);  
                for i = n-1:-1:1
                    result = result * (x - xa(i)) + da(i); 
                end
            end

            function result = NewtonInterpolation(xa, ya, x)
                da = DividedDifference(xa, ya);
                result = NewtonForm(xa, da, x);
            end

            KquaNS = NewtonInterpolation(xa, ya, x1);

            x = sym('x');
            DaThucNS = NewtonInterpolation(xa, ya, x);
            DaThucNS = vpa(simplify(DaThucNS), 4); 

            app.KetQuaNoiSuyLabel.Text = sprintf('Kết quả nội suy: %.4f', KquaNS);
            app.KetQuaDaThucNoiSuyLabel.Text = sprintf('Đa thức nội suy: %s', char(DaThucNS));

            cla(app.UIAxesNS);
            x_vals = linspace(min(xa), max(xa), 100); 
            y_vals = arrayfun(@(x_val) NewtonInterpolation(xa, ya, x_val), x_vals); 
            plot(app.UIAxesNS, xa, ya, 'ro', 'MarkerFaceColor', 'r'); % Dữ liệu thực với điểm đỏ
            hold(app.UIAxesNS, 'on');
            plot(app.UIAxesNS, x_vals, y_vals, 'b-', 'LineWidth', 1.5); % Đồ thị hàm nội suy với đường xanh

        end

        % Button pushed function: TuyenTinhButton
        function TuyenTinhButtonPushed(app, event)
            x = app.X_EditField_2.Value;
            y = app.Y_EditField_2.Value;
            x = str2num(x);
            y = str2num(y);
            
            function [a1, a0, r2] = HoiQuyTuyenTinh(x, y)
            n = length(x); 
            sumx = 0; 
            sumy = 0; 
            sumxy = 0;
            sumx2 = 0;
            st = 0;  
            sr = 0; 
            for i = 1:n
                sumx = sumx + x(i); 
                sumy = sumy + y(i); 
                sumxy = sumxy + x(i)*y(i); 
                sumx2 = sumx2 + x(i)*x(i); 
            end 
            xm = sumx/n; 
            ym = sumy/n; 
            a1 = (n*sumxy - sumx*sumy)/(n*sumx2 - sumx*sumx); 
            a0 = ym - a1*xm; 
            for i = 1: n
                st = st + (y(i) - ym).^2; 
                sr = sr + (y(i) - a1*x(i) - a0).^2; 
            end 
            r2 = (st - sr)/st; 
            end 

            [a1, a0, r2] = HoiQuyTuyenTinh(x, y);

            x_DuDoan = app.GiaTriDuDoanEditField.Value;
            y_DuDoan = a0 + a1*x_DuDoan;        
            app.KetQuaDuDoanLabel.Text = sprintf('Kết quả dự đoán: %.4f', y_DuDoan);
            app.KetQuaPtHoiQuyLabel.Text = sprintf('Kết quả phương trình hồi quy: y = %.4f + %.4f*x', a0, a1);
           
            cla(app.UIAxes_2);
            hold(app.UIAxes_2, 'on');
            scatter(app.UIAxes_2, x, y, 'r', 'filled'); 
            y_fit = a1 * x + a0; 
            plot(app.UIAxes_2, x, y_fit, '-b', 'LineWidth', 2);
            legend(app.UIAxes_2, 'Dữ liệu thực', 'Đường hồi quy');
            grid(app.UIAxes_2, 'on');
            hold(app.UIAxes_2, 'off');
        end

        % Button pushed function: KetQua
        function KetQuaButtonPushed(app, event)
             % Lấy dữ liệu từ giao diện
    x = app.NhapX.Value; % Dữ liệu x 
    y = app.NhapY.Value; % Dữ liệu y 
    h = str2double(app.NhapH.Value);           % Bước h
    hs = app.NhapHSf.Value;                     % Hàm số 
    value = str2double(app.NhapXcantim.Value); % Giá trị cần tính
    pp = app.ChonPP.Value;                     % Phương pháp đạo hàm
    saiso = app.ChonSaiso.SelectedObject.Text;  % Giá trị sai số

% Kiểm tra điều kiện đầu vào
    if isempty(x) && isempty(y) && ~isempty(hs) && ~isempty(h) && ~isempty(value)
        % Xử lý dữ liệu hàm số
        try
            f = str2func(['@(x)', hs]); % Chuyển chuỗi thành hàm số
        catch
            app.KetQuaLabel.Text = 'Lỗi: Hàm số không hợp lệ.';
            return;
        end
    
        % Tính đạo hàm theo sai số và phương pháp
        switch saiso
            case 'O(h^2)'
                if strcmp(pp, 'Trung tâm')
                    result = (f(value + h) - f(value - h)) / (2*h);
                elseif strcmp(pp, 'Tiến')
                    result = (-3*f(value) + 4*f(value + h) - f(value + 2*h)) / (2*h);
                else
                    result = (3*f(value) - 4*f(value - h) + f(value - 2*h)) / (2*h);
                end
            case 'O(h)'
                if strcmp(pp, 'Tiến')
                    result = (-1*f(value) + f(value + h)) / (h);
                else
                    result = (f(value) - f(value - h)) / (h);
                end
    
            otherwise
                app.KetQuaLabel.Text = 'Lỗi: Sai số không hợp lệ.';
                return;
        end
    
        % Hiển thị kết quả
        app.KetQuaLabel.Text = sprintf('Kết quả: %.4f', result);
        return;
    
    elseif ~isempty(x) && ~isempty(y) && isempty(hs) && ~isempty(h) && ~isempty(value)
        % Xử lý bảng giá trị x và y
        x = str2double(strsplit(x));  % Chuyển đổi x thành mảng số 
        y = str2double(strsplit(y));  % Chuyển đổi y thành mảng số 
        h = x(2) - x(1);
        % Kiểm tra x và y có cùng độ dài không
        if ~isempty(x) && ~isempty(y) && length(x) ~= length(y)
            app.KetQuaLabel.Text = 'Lỗi: x và y phải có cùng độ dài.';
            return;
        end
        n = (value - x(1))/h + 1;
        n = cast(n,'uint16');
        % Tính đạo hàm theo sai số và phương pháp
        switch saiso
            case 'O(h^2)'
                if strcmp(pp, 'Trung tâm')
                    result = (y(n + 1) - y(n - 1)) / (2*h);
                elseif strcmp(pp, 'Tiến')
                    if n >= length(x)-1
                        app.KetQuaLabel.Text = 'Không thể sử dụng xấp xỉ tiến.';
                        return;
                    end
                    result = (-3*y(n) + 4*y(n + 1) - y(n + 2)) / (2*h);
                else 
                    if n <= 2
                        app.KetQuaLabel.Text = 'Không thể sử dụng xấp xỉ lùi.';
                        return;
                    end
                    result = (3*y(n) - 4*y(n - 1) + y(n - 2)) / (2*h);
                end
    
            case 'O(h)'
                if strcmp(pp, 'Tiến')
                    if n >= length(x)
                        app.KetQuaLabel.Text = 'Không thể sử dụng xấp xỉ tiến.';
                        return;
                    end
                    result = (-y(n) + y(n + 1)) / (h);
                else
                    if n <= 1
                        app.KetQuaLabel.Text = 'Lỗi: Không thể sử dụng xấp xỉ lùi.';
                        return;
                    end
                    result = (y(n) - y(n - 1)) / (h);
                end
        end
    
        % Hiển thị kết quả
        app.KetQuaLabel.Text = sprintf('Kết quả: %.4f', result);
    
    else
        app.KetQuaLabel.Text =  'Lỗi: Vui lòng nhập dữ liệu đầu vào.';
    end
        end

        % Button pushed function: resultButton
        function resultButtonPushed(app, event)
             % Lấy dữ liệu từ giao diện
    method = app.methodDropDown.Value;                  % Phương pháp tích phân
    a = str2double(app.lowerBoundInput.Value);          % Cận dưới
    b = str2double(app.upperBoundInput.Value);          % Cận trên
    n = str2double(app.nInput.Value);                   % Số đoạn

    if ~isempty(app.xInput.Value) && ~isempty(app.yInput.Value)
        % Nếu nhập dữ liệu x và y
        x = str2num(app.xInput.Value); % Chuyển chuỗi thành mảng
        y = str2num(app.yInput.Value);

        % Kiểm tra và xử lý theo phương pháp đã chọn
        switch method
            case 'Hình thang'
                h = diff(x);
                result = sum(h .* (y(1:end-1) + y(2:end)) / 2); % Công thức hình thang
            case 'Simpson 1/3'
                n = length(x) - 1;
                if mod(n, 2) ~= 0
                    result = 'Số đoạn phải là số chẵn!';
                    app.resultField.Value = result;
                    return;
                end
                h = (x(end) - x(1)) / n;
                result = (h / 3) * (y(1) + y(end) + 4 * sum(y(2:2:end-1)) + 2 * sum(y(3:2:end-2)));
            case 'Simpson 3/8'
                n = length(x) - 1;
                if mod(n, 3) ~= 0
                    result = 'Số đoạn phải chia hết cho 3!';
                    app.resultField.Value = result;
                    return;
                end
                h = (x(end) - x(1)) / n;
                result = (3 * h / 8) * (y(1) + y(end) + 3 * sum(y(2:3:end-1) + y(3:3:end-2)) + 2 * sum(y(4:3:end-3)));
        end
    elseif ~isempty(app.functionInput.Value)
        % Nếu nhập hàm số
        func = str2func(['@(x)' app.functionInput.Value]); % Chuyển chuỗi thành hàm
        x = linspace(a, b, n+1);
        y = arrayfun(func, x);

        % Kiểm tra và xử lý theo phương pháp đã chọn
        switch method
            case 'Hình thang'
                h = diff(x);
                result = sum(h .* (y(1:end-1) + y(2:end)) / 2); % Công thức hình thang
            case 'Simpson 1/3'
                if mod(n, 2) ~= 0
                    result = 'Số đoạn phải là số chẵn!';
                    app.resultField.Value = result;
                    return;
                end
                h = (x(end) - x(1)) / n;
                result = (h / 3) * (y(1) + y(end) + 4 * sum(y(2:2:end-1)) + 2 * sum(y(3:2:end-2)));
            case 'Simpson 3/8'
                if mod(n, 3) ~= 0
                    result = 'Số đoạn phải chia hết cho 3!';
                    app.resultField.Value = result;
                    return;
                end
                h = (x(end) - x(1)) / n;
                result = (3 * h / 8) * (y(1) + y(end) + 3 * sum(y(2:3:end-1) + y(3:3:end-2)) + 2 * sum(y(4:3:end-3)));
        end
    else
        % Nếu không nhập đủ dữ liệu
        result = 'Vui lòng nhập đủ dữ liệu!';
    end

    % Hiển thị kết quả
    app.resultField.Value = num2str(result);
        end
    end
